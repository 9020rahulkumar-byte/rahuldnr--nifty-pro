// -- TOTP Generator (no extra lib needed) ---------------------
function generateTOTP(secret) {
  // If user provides static TOTP pin instead of secret, return it directly
  if (secret && secret.length <= 6 && /^\d+$/.test(secret)) return secret;
  // Return "000000" as fallback Ś user should paste fresh TOTP manually
  return secret || "000000";
}

// -- Login to Angel One ----------------------------------------
async function loginAngel() {
  try {
    const totp = generateTOTP(CONFIG.totpSecret);
    const res = await fetch(${BASE}/rest/auth/angelbroking/user/v1/loginByPassword, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Accept": "application/json",
        "X-UserType": "USER",
        "X-SourceID": "WEB",
        "X-ClientLocalIP": "127.0.0.1",
        "X-ClientPublicIP": "127.0.0.1",
        "X-MACAddress": "AA:BB:CC:DD:EE:FF",
        "X-PrivateKey": CONFIG.apiKey,
      },
      body: JSON.stringify({
        clientcode: CONFIG.clientCode,
        password:   CONFIG.password,
        totp:       totp,
      }),
    });
    const data = await res.json();
    if (data.status && data.data) {
      authToken    = data.data.jwtToken;
      refreshToken = data.data.refreshToken;
      lastLogin    = Date.now();
      console.log("? Angel One Login SUCCESS");
      return true;
    } else {
      console.error("? Login failed:", data.message || JSON.stringify(data));
      return false;
    }
  } catch (e) {
    console.error("? Login error:", e.message);
    return false;
  }
}

// -- Get LTP (Last Traded Price) -------------------------------
async function getLTP(symbol) {
  try {
    const info = SYMBOL_TOKENS[symbol];
    if (!info) return null;
    const data = await angelPOST("/rest/secure/angelbroking/market/v1/quote/", {
      mode: "FULL",
      exchangeTokens: { [info.exch]: [info.token] },
    });
    if (data.status && data.data?.fetched?.length > 0) {
      const q = data.data.fetched[0];
      return {
        symbol,
        ltp:        q.ltp,
        open:       q.open,
        high:       q.high,
        low:        q.low,
        close:      q.close,
        change:     parseFloat(((q.ltp - q.close) / q.close * 100).toFixed(2)),
        changeAmt:  parseFloat((q.ltp - q.close).toFixed(2)),
        volume:     q.tradedVolume,
        avgPrice:   q.avgPrice,
        upperCircuit: q.upperCircuit,
        lowerCircuit: q.lowerCircuit,
      };
    }
    return null;
  } catch(e) {
    console.error("getLTP error:", e.message);
    return null;
  }
}

// -- Get OHLC Candles ------------------------------------------
async function getCandles(symbol, interval, fromDate, toDate) {
  try {
    const info = SYMBOL_TOKENS[symbol];
    if (!info) return [];
    // interval: ONE_MINUTE, FIVE_MINUTE, FIFTEEN_MINUTE, ONE_HOUR, ONE_DAY
    const data = await angelPOST("/rest/secure/angelbroking/historical/v1/getCandleData", {
      exchange:    info.exch,
      symboltoken: info.token,
      interval,
      fromdate:    fromDate,
      todate:      toDate,
    });
    if (data.status && data.data) {
      return data.data.map(c => ({
        time: c[0],
        o: c[1], h: c[2], l: c[3], c: c[4], v: c[5],
      }));
    }
    return [];
  } catch(e) {
    console.error("getCandles error:", e.message);
    return [];
  }
}

// -- Next Thursday (expiry) ------------------------------------
function getNextThursday() {
  const d = new Date();
  const day = d.getDay();
  const daysUntilThursday = (4 - day + 7) % 7 || 7;
  d.setDate(d.getDate() + daysUntilThursday);
  return ${d.getDate().toString().padStart(2,"0")}${d.toLocaleString("en-US",{month:"short"}).toUpperCase()}${d.getFullYear()};
}

// -- Date helpers for candle fetch -----------------------------
function candleDateRange(tf) {
  const now   = new Date();
  const to    = formatDate(now);
  const from  = new Date(now);
  const tfMap = {
    "FIVE_MINUTE":    { days: 5  },
    "FIFTEEN_MINUTE": { days: 10 },
    "ONE_HOUR":       { days: 30 },
    "FOUR_HOUR":      { days: 90 },
    "ONE_DAY":        { days: 365},
  };
  from.setDate(from.getDate() - (tfMap[tf]?.days || 10));
  return { from: formatDate(from), to };
}

function formatDate(d) {
  return ${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,"0")}-${String(d.getDate()).padStart(2,"0")} ${String(d.getHours()).padStart(2,"0")}:${String(d.getMinutes()).padStart(2,"0")};
}

// -- Cache (to avoid rate limits) ------------------------------
const cache = {};
function cached(key, ttlMs, fn) {
  const now = Date.now();
  if (cache[key] && (now - cache[key].time) < ttlMs) return Promise.resolve(cache[key].data);
  return fn().then(data => { cache[key] = { data, time: now }; return data; });
}

  const msg = JSON.stringify(data);
  clients.forEach(ws => {
    if (ws.readyState === WebSocket.OPEN) ws.send(msg);
  });
}

// Push live data every 3 seconds to all connected dashboards

}
